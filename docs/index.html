Простой аккордион - возможные опции ниже

jQuery('ul.accordion').slideAccordion({
						activeClass:'open',      // класс "active" открытого состояния
						opener:'>a.opener',      // селектор ссылки-opener
						slider:'>div.slide',     // селектор дропа аккордиона
						collapsible:false,       // разрешать сворачивать активный элемент аккордиона
						animSpeed: 300,          // скорость анимации аккордиона
						scrollToActiveItem: {    // прокрутка к активному элементу аккордиона
							enable: true,
							breakpoint: 1920, // max-width
							animSpeed: 600,
							extraOffset: null
						}});
